```js
// search one word.
idx.search('foo')
// search two word using 'or'.
idx.search('foo bar')
// wildcard
idx.search('foo*')
// field
idx.search('title:foo')
// boost
idx.search('foo^10 bar')
// fuzzy (내용중 숫자만큼 달라도 됨).
idx.search('foo~1')
// 반드시 있음/없음.
idx.search("+foo bar -baz")
```
